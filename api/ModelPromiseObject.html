ModelPromiseObject Class

Extends: PromiseObject
Defined in: src/data/promise_object.js:25

Acts just like PromiseObject only it's able to hold the ID and typeKey of a model before it's resolved completely.

var user = EG.ModelPromiseObject.create({
    promise: this.store.find('user', '52'),
    id: '52',
    typeKey: 'user'
});

user.get('isPending'); // true
user.get('id'); // '52'
user.get('typeKey'); // 'user'
